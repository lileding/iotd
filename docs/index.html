Quick Start

# Download and extract
wget https://github.com/lileding/iotd/releases/download/v0.2.0/iotd-linux-x86_64.tar.gz
tar -xzf iotd-linux-x86_64.tar.gz

# Run the server
./iotd-linux-x86_64

# Or with custom configuration
./iotd-linux-x86_64 -c config.toml
